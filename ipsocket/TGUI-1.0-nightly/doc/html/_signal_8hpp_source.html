131 template <typename Func, typename... BoundArgs, typename std::enable_if_t<std::is_convertible<Func, std::function<void(const BoundArgs&...)>>::value>* = nullptr>
132 unsigned int connect(const Func& func, const BoundArgs&... args)
133 {
134 const auto id = ++m_lastSignalId;
135#if defined(__cpp_if_constexpr) && (__cpp_if_constexpr >= 201606L)
136 if constexpr(sizeof...(BoundArgs) == 0)
137 m_handlers[id] = func;
138 else
139#endif
140 {
141 m_handlers[id] = [=]{ invokeFunc(func, args...); };
142 }
143
144 return id;
145 }

241 template <typename Type>

248#if defined(__cpp_lib_invoke) && (__cpp_lib_invoke >= 201411L)
249 template <typename Func, typename... Args>
250 static void invokeFunc(Func&& func, Args&&... args)
251 {
252 std::invoke(std::forward<Func>(func), std::forward<Args>(args)...);
253 }
254#else
255 // std::invoke only exists in c++17 so we use our own implementation to support c++14 compilers
256 template <typename Func, typename... Args, typename std::enable_if_t<std::is_member_pointer<std::decay_t<Func>>::value>* = nullptr>
257 static void invokeFunc(Func&& func, Args&&... args)
258 {
259 (std::mem_fn(func))(std::forward<Args>(args)...);
260 }
261
262 template <typename Func, typename... Args, typename std::enable_if_t<!std::is_member_pointer<std::decay_t<Func>>::value>* = nullptr>
263 static void invokeFunc(Func&& func, Args&&... args)
264 {
265 std::forward<Func>(func)(std::forward<Args>(args)...);
266 }
267#endif
